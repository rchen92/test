ASM  
将存在的数据文件迁移到ASM，将ASM的数据文件迁移出ASM
	使用rman的方式
		set newname datafile <'olddir'> to <'newdir'>;
		restore datafile <>;
		recover datafile <>;
		switch datafile <>;

#例：将users表空间文件 --->  ASM磁盘
RMAN> BACKUP TABLESPACE users;
RMAN> run {
	2 sql 'alter tablespace users offline immediate';
	3 set newname for datafile 4 to '+DATA';
	4 restore tablespace users;
	5 switch datafile 4;
	6 recover tablespace users;
	7 sql 'alter tablespace users online';
	8 }
#例：将users表空间文件 --->  系统存储
RMAN> run {
2> sql 'alter tablespace users offline immediate';
3> set newname for datafile 4 to '/u01/app/oracle/oradata/orcl/users01.dbf';
4> restore datafile 4;
5> switch datafile 4;
6> recover datafile 4;
7> sql 'alter tablespace users online';
8> }
