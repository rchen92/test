表空间传输
	比如说一个表很大，要将这个表同步到另外一台电脑上去。用户也很多，进行迁移也很困难
orcl库和orcl1库，将orcl库中的tt表空间直接放到orcl1中，并且将其中的数据也放到orcl1中

[oracle@oracle ~]$ echo $ORACLE_SID
orcl
[oracle@oracle ~]$ sqlplus / as sysdba
SQL> create tablespace tt datafile '/u01/app/oracle/oradata/orcl/tt.dbf' size 10m;		#创建实验所用的表空间tt
SQL> create user tt identified by tt default tablespace tt;			#创建用户tt，默认表空间为tt
SQL> grant connect,resource to tt;						#授予权限
SQL> create table  tt.e as select * from scott.emp;		#创建tt用户下的表
SQL> alter tablespace tt read only;				#将tt表空间改为只读
[oracle@oracle ~]$ exp \'sys/oracle as sysdba\' file=/backup/tt.dmp transport_tablespace=y tablespaces=tt

[oracle@oracle ~]$ export ORACLE_SID=orcl1
[oracle@oracle ~]$ sqlplus / as sysdba		#需要先创建相同的用户，并且两个数据库的字符集要一样
